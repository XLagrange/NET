vector<int> postorderTraversal(TreeNode *root) {
    vector<int> res;
    if (root == NULL) {
        return res;
    }

    TreeNode *cur = root;
    stack<TreeNode*> s;
    while (cur != NULL || !s.empty()) {
        if (cur != NULL) {
            if (cur->right != NULL) {
                s.push(cur->right);
            }
            s.push(cur);
            cur = cur->left;
        } else {
            cur = s.top();
            s.pop();
            if (!s.empty() && cur->right == s.top()) {
                TreeNode *right_child = s.top();
                s.pop();
                s.push(cur);
                cur = right_child;
            } else {
                res.push_back(cur->val);
                cur = NULL;
            }
        }
    }
    return res;
}
